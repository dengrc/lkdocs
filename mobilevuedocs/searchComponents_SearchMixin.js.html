define([
    "dojo/text!./search.html"
], function (template) {
	/**
	 * @author 邓如春 <1107151332@qq.com>
	 * @module SearchMixin
	 * @mixin
	 * 
	 * @description
	 * ### [插槽](https://cn.vuejs.org/v2/guide/components-slots.html)
	 * - right
	 * - fill
	 * 
	 * ##### Search 控件中的共用方法、属性
	 * #### Props 接受父组件的传值
	 * 
	 * @prop	{Boolean}		[arrow=true]			向右箭头
	 * @prop	{Boolean}		[button=true]			按钮一样的点击效果
	 * @prop	{String}		[label]					显示在左侧的文字
	 * @prop	{String}		[placeholder="请选择"]	插槽 right 空时，显示的文字
	 * @prop	{Boolean}		[isJoin=true]			多选时，是否链接成字符串
	 * @prop	{String}		[separator=","]			链字符串的分割符
	 * @prop	{String}		[idProperty="id"]		主键字段
	 * @prop	{Boolean}		[multiple=false]		多选
	 * @prop	{Boolean}		[readonly=false]		只读
	 * @prop	{Boolean}		[disable=false]			禁用
	 * @prop	{Object}		[searchMap={}]			字段映射
	 * @prop	{Object}		[searchParam={}]		打开 search 页面时的参数
	 * @prop	{Object}		[defaultValues={}]		默认值
	 * @prop	{Object}		[context={}]			操作后值映射到的对象
	 */
    return {
        props: {
            arrow: {
                type: Boolean,
                "default": true
            },
            button: {
                type: Boolean,
                "default": true
            },
            label: {
                type: String,
                "default": ""
            },
            placeholder: {
                type: String,
                "default": "请选择"
            },
            isJoin: {
                type: Boolean,
                "default": true
            },
            separator: {
                type: String,
                "default": ","
            },
            idProperty: {
                type: String,
                "default": "id"
            },
            multiple: {
                type: Boolean
            },
            readonly: {
                type: Boolean
            },
            disable: {
                type: Boolean
            },
            searchMap: {
                type: Object,
                "default": function () {
                    return {}
                }
            },
            searchParam: {
                type: Object,
                "default": function () {
                    return {}
                }
            },
            defaultValues: {
                type: Object,
                "default": function () {
                    return {}
                }
            },
            context: {
                type: Object,
                "default": function () {
                    return {}
                }
            }
        },
        computed: {
            idKey: function () {
                if (!this.idProperty) {
                    return null
                }
                var map = this.getSearchMap();
                var id = null;
                Object.keys(map).some(function (key) {
                    if (map[key] == this.idProperty) {
                        id = key;
                        return true
                    }
                }, this);
                return id
            }
        },
        methods: {
            onClick: function () {
                if (this.readonly || this.disable)
                    return;
                this.openSearchView();
            },
            openSearchView: function () {
                this.$root.startView(this.searchView, {
                    _onOk: this.onSearchResult,
                    param: this.getSearchParam()
                })
            },
            onSearchResult: function (result) {
                var oid = this.context[this.idKey];
                if (result) {
                    var map = this.getSearchMap(),
                        mapKeys = Object.keys(map),
                        values = {};
                    //创建键值对
                    mapKeys.forEach(function (key) {
                        values[key] = [];
                    });
                    //映射值
                    (result instanceof Array ? result : [result]).forEach(function (item) {
                        mapKeys.forEach(function (key) {
                            values[key].push(item[map[key]])
                        }, this)
                    }, this);
                    //连接值
                    if (this.isJoin) {
                        mapKeys.forEach(function (key) {
                            values[key] = values[key].join(this.separator);
                        });
                    }
                    //附加到数据上下文
                    this.mixinsContext(values)
                } else {
                    this.resetValues()
                }
                var id = this.context[this.idKey];
                if (oid !== id) {
                    this.$emit("change", {
                        oid: oid,
                        id: id,
                        result: result
                    })
                }
            },
            resetValues: function () {
                var defaultValues = this.getDefaultValues(),
                    map = this.getSearchMap();
                var values = {};
                if (map) {
                    for (var key in map) {
                        values[key] = defaultValues[map[key]];
                    }
                } else {
                    values = defaultValues;
                }
                this.mixinsContext(values);
            },
            mixinsContext: function (data) {
                Object.keys(data).forEach(function (key) {
                    if (key in this.context) {
                        if (data[key] != undefined) {
                            this.context[key] = data[key]
                        }
                    }
                }, this)
            },
            getSearchParam: function () {
                return this.searchParam
            },
            getDefaultValues: function () {
                return this.defaultValues
            },
            getSearchMap: function () {
                return this.searchMap
            }
        },
        template: template
    }
})
