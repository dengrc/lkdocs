define([
    "dojo/_base/declare",
    "mobile/store/config",
    "mobile/utils/formatter",
    "mobile/utils/DateHelper",
    "../QueryBase"
], function (declare, config, formatter, DateHelper, QueryBase) {
    return declare([QueryBase], {
        init: function () {
            this.inherited(arguments);
            this.vueContent = this.addVue({
                el: this.vuePageEl,
                mixins: QueryBase.vueMixins,
                data: function () {
                    //var productVersion = config.getCurrent().userinfo.ProductVersion;
                    var todaydate = formatter.dateFormat(new Date(), 'yyyy-MM-dd');
                    var thismonth = formatter.dateFormat(new Date(), 'yyyy-MM');
                    return {
                        title: "销售日月报",
                        filterName: "SaleDailyReportFilter",
                        maxDate: todaydate,
                        maxMonth: thismonth,
                        sumData: {
                            salediscountmoney: 0,
                            costmoney: 0,
                            profit: 0,
                            profitrate:0
                        },
                        "reportparam": {
                            "pagerow": 20,
                            "pagenumber": 0,
                            "pagetype": "erp",
                            "orders": 'billcode desc'
                        },

                        "conditionparam": {
                            "billcode": "",
                            "todaydate": todaydate,
                            "thismonth": thismonth,
                            "begindate": DateHelper.getDateRange("thisMonth").begindate,
                            "enddate": DateHelper.getDateRange("thisMonth").enddate,
                            "ifred": false,
                            "billtype": "2005",
                            "sinoutid": "",
                            "sinoutfullname": "",
                            "bid": "",
                            "bfullname": "",
                            "pageid": "sale_day",
                            "cid": "",
                            "cfullname": "",
                            "billtypes": "-1"
                        },
                        tabValue: 0,
                        tabs: [{
                            title: '日',
                            value: 0
                        }, {
                            title: '月',
                            value: 1
                        }],
                        sortOptions: [
                            {
                                value: 'billcode asc',
                                label: '单据编号升序'
                            },
                            {
                                value: 'billcode desc',
                                label: '单据编号倒序'
                            },
                            {
                                value: 'salediscountmoney asc',
                                label: '销售金额升序'
                            },
                            {
                                value: 'salediscountmoney desc',
                                label: '销售金额倒序'
                            }
                        ],
                        sortOptions1: [
                            {
                                value: 'billdate desc',
                                label: '时间倒序'
                            },
                            {
                                value: 'billdate asc',
                                label: '时间升序'
                            },
                            {
                                value: 'salediscountmoney desc',
                                label: '销售金额由高到低'
                            },
                            {
                                value: 'salediscountmoney asc',
                                label: '销售金额由低到高'
                            }
                        ]
                    }
                },
                methods: {
                    onTabChange: function (tab) {
                        if (tab.value == 1) {
                            this.reportparam.orders = "billdate desc";
                            this.conditionparam.thismonth = formatter.dateFormat(new Date(), 'yyyy-MM');
                            this.filterName = "SaleMonthlyReportFilter";
                            this.conditionparam.pageid = "sale_month";
                            this.conditionparam.billtype = "2004";
                            this.conditionparam.begindate = DateHelper.getDateRange("currentMonth", this.conditionparam.thismonth).begindate;
                            this.conditionparam.enddate = DateHelper.getDateRange("currentMonth", this.conditionparam.thismonth).enddate;
                            delete this.conditionparam['todaydate'];
                        } else {
                            this.filterName = "SaleDailyReportFilter";
                            this.conditionparam.pageid = "sale_day";
                            this.conditionparam.billtype = "2005";
                            this.conditionparam.todaydate = formatter.dateFormat(new Date(), 'yyyy-MM-dd');
                            this.reportparam.orders = "billcode desc";
                            delete this.conditionparam['begindate'];
                            delete this.conditionparam['enddate'];
                        }
                        this.conditionparam.billcode = "";
                        //this.resetSortOptions(tab.value);
                        this.doQuery();
                    },
                    //resetSortOptions: function (tabType) {
                    //    this.sortOptions = [{
                    //        value: 0,
                    //        label: '默认排序'
                    //    },
                    //    {
                    //        value: 'billcode desc',
                    //        label: '订单编号'
                    //    },
                    //    {
                    //        value: 'salediscountmoney desc',
                    //        label: '销售金额'
                    //    },
                    //    {
                    //        value: 'cfullname desc',
                    //        label: '经手人'
                    //    },
                    //    {
                    //        value: 'bfullname asc',
                    //        label: '部门'
                    //    }
                    //    ]
                    //    if (tabType == 0) {
                    //        this.sortOptions = [{

                    //            value: 0,
                    //            label: '默认排序'
                    //        },
                    //        {
                    //            value: 'salediscountmoney desc',
                    //            label: '销售金额'
                    //        }
                    //        ];
                    //    }
                    //},
                    dataHandler: function (data) {
                        data.datasource.forEach(function (item) {
                            item._expand = false
                        });
                        return data
                    },
                    expandItem: function (data) {
                        data._expand = !data._expand;
                        this.$refs.recyclist.resize();
                    },
                    /**
                     * 销售月报切换月份
                     * **/
                    onMonthChanged: function (data) {
                        this.conditionparam.begindate = DateHelper.getDateRange("currentMonth", this.conditionparam.thismonth).begindate;
                        this.conditionparam.enddate = DateHelper.getDateRange("currentMonth", this.conditionparam.thismonth).enddate;
                        this.doQuery();
                    },
                    onItemClick: function (_data) {
                        var data = Object.assign({}, _data);
                        //日报打开单据
                        if (this.tabValue == 0) {
                            data.billtype == 903 && this.startView("CostBill", data) ||
                                data.billtype == 602 && this.startView("SaleReturnsBill", data) ||
                                data.billtype == 607 && this.startView("RetailBill", data) ||
                                data.billtype == 801 && this.startView("PayByCustomerBill", data) ||
                                data.billtype == 904 && this.startView("OtherInComeBill", data) ||
                                data.billtype == 601 && this.startView("SaleBill", data) ||
                                data.billtype == 603 && this.startView("SaleExchangeBill", data);
                        } else {//月报打开销售明细
                            this.startView("SaleReport", { begindate: data.billdate, enddate: data.billdate, ifred: this.conditionparam.ifred, cid: this.conditionparam.cid, cfullname: this.conditionparam.cfullname, bid: this.conditionparam.bid, bfullname: this.conditionparam.bfullname, sid: this.conditionparam.sinoutid, sfullname: this.conditionparam.sinoutfullname });
                        }
                    }
                }
            })
        }
    })
});
